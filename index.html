تسجيل دخول الموظفين

function login() { const email = document.getElementById('email').value; const password = document.getElementById('password').value; firebase.auth().signInWithEmailAndPassword(email, password) .then((userCredential) => { window.location.href = "home.html"; // التوجه للصفحة الرئيسية }) .catch((error) => { document.getElementById('message').innerText = "خطأ في البيانات أو غير مصرح لك"; }); }
